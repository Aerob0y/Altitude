# Altitude
Altitude/
├─ app.R                          # very thin launcher (sources R/ and calls run_app())
├─ R/                             # all R code lives here
│  ├─ mod_ui_server.R             # Shiny modules (one file per module or per feature)
│  ├─ utils_data.R                # data helpers (wrangle, validate)
│  ├─ utils_plot.R                # plotting helpers (ggplot/plotly builders)
│  ├─ utils_theme.R               # shared theming, CSS class helpers
│  └─ run_app.R                   # run_app() that builds UI & server from modules
├─ data-raw/                      # scripts to build packaged datasets
│  └─ rbnz_build.R                # pulls/cleans RBNZ data, writes to data/
├─ data/                          # R-ready .rds or .qs artifacts (not raw files)
├─ inst/
│  └─ app/www/                    # static assets served by Shiny
│     ├─ css/                     # styles moved from /styles
│     ├─ js/
│     └─ img/
├─ tests/
│  ├─ testthat/                   # unit tests for helpers + module logic
│  └─ shinytest2/                 # snapshot tests for the app
├─ man/                           # (auto) if you roxygenise helpers; optional
├─ DESCRIPTION                    # treat app “as a package” (simplifies deps)
├─ NAMESPACE                      # generated by roxygen2; optional but nice
├─ renv.lock                      # captured environment
├─ .lintr                         # you already have this — keep it
├─ .Rbuildignore                  # ignore inst/app/www dev bits, etc.
├─ .gitignore
└─ README.md
